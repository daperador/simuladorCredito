CREATE TABLE ADMINISTRADOR 
(
  ID NUMBER NOT NULL 
, NOMBRES VARCHAR2(20) 
, APELLIDOS VARCHAR2(20) 
, EMAIL VARCHAR2(50) 
, CONTRASENA VARCHAR2(50) 
, URL VARCHAR2(500) 
, CONSTRAINT ADMINISTRADOR_PK PRIMARY KEY 
  (
    ID 
  )
  ENABLE 
);

CREATE TABLE LINEA 
(
  ID NUMBER NOT NULL 
, ID_ADMON NUMBER 
, NOMBRE VARCHAR2(50) 
, TASA FLOAT 
, CONSTRAINT LINEA_PK PRIMARY KEY 
  (
    ID 
  )
  ENABLE 
);

CREATE TABLE PLAN_PAGO 
(
  ID NUMBER NOT NULL 
, FECHA_CREACION DATE 
, DOCUMENTO VARCHAR2(20) 
, VALOR NUMBER 
, PLAZO NUMBER 
, ID_LINEA NUMBER 
, ESTADO VARCHAR2(20) 
, NIVEL_RIESGO FLOAT
, FECHA_NACIMIENTO DATE
, FECHA_MODIFICACION DATE
, CONSTRAINT PLAN_PAGO_PK PRIMARY KEY 
  (
    ID 
  )
  ENABLE 
);



CREATE TABLE CUOTA 
(
  ID NUMBER NOT NULL 
, ID_PLAN NUMBER 
, NUMERO_CUOTA NUMBER 
, INTERESES DOUBLE PRECISION 
, CAPITAL DOUBLE PRECISION 
, TOTAL DOUBLE PRECISION 
, SALDO DOUBLE PRECISION 
, CONSTRAINT CUOTA_PK PRIMARY KEY 
  (
    ID 
  )
  ENABLE 
);



ALTER TABLE PLAN_PAGO
ADD CONSTRAINT PLAN_PAGO_FK1 FOREIGN KEY
(
  ID_LINEA 
)
REFERENCES LINEA
(
  ID 
)
ENABLE;


ALTER TABLE LINEA
ADD CONSTRAINT LINEA_FK1 FOREIGN KEY
(
  ID_ADMON 
)
REFERENCES ADMINISTRADOR
(
  ID 
)
ENABLE;


CREATE SEQUENCE SEQ_ADMINISTRADOR INCREMENT BY 1 START WITH 1 MINVALUE 1;
CREATE SEQUENCE SEQ_CUOTA INCREMENT BY 1 START WITH 1 MINVALUE 1;
CREATE SEQUENCE SEQ_LINEA INCREMENT BY 1 START WITH 1 MINVALUE 1;
CREATE SEQUENCE SEQ_PLAN INCREMENT BY 1 START WITH 1 MINVALUE 1;
